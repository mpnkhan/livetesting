//Spread operator a.k.a. "Three-dots" tricks <3

//convert a string to array
const str = 'kitty';
const newArr = [...str]; 
console.log(newArr);   // ['k', 'i', 't', 't', 'y'];

//concat two arrays (yeah, con"cat"!)
const arr1 = ['solid', 'bicolor', 'tabby'];
const arr2 = ['calico', 'tortoiseshell'];
const concatArr = [...arr1, ...arr2]; 
console.log(concatArr)   // ['solid', 'bicolor', 'tabby', 'calico', 'tortoiseshell']


//find the largest
const catWeight = [10, 9, 6, 12];
const max = Math.max(...catWeight); 
console.log(max);   // 12

//cloning
const allCatNames = ['chewie', 'leia', 'yoda', 'chewie', 'luke', 'leia'];
const allCatNamesCopy = [...allCatNames];

//if you want to copy sans duplicates, use the Set object!
const catNames = [...new Set(allCatNames)]; //['chewie', 'leia', 'yoda', 'luke'];
console.log(catNames)
